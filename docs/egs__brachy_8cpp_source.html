86void printParticleWithSpherical(EGS_Particle p) {
87 EGS_Float x,y,z, r, R, theta, phi;
88
89 x = p.x.x;
90 y = p.x.y;
91 z = p.x.z;
92
93 r = sqrt(x*x + y*y);
94 R = sqrt(x*x + y*y + z*z);
95 theta = acos(z/R)*180/M_PI;
96 phi = atan2(y, x)*180/M_PI;
97
98 PRINT_PARTICLE_WITH_DIR(p);
99 cout << " R = " << R << endl;
100 cout << " r = " << r << endl;
101 cout << " z = " << z << endl;
102 cout << " theta (deg) = " << theta << endl;
103 cout << " phi (deg) = " << phi << endl;
104
105}

302int EB_Application::initGeometry() {
303 // override default initGeometry so we can manually create our own geometry
304 // which will allow us to track region numbers for each geometry object individually
305
306 timing_blocks.addTimer("egs_brachy::initGeometry");
307
308 EGS_Input *ginput = input->getInputItem("geometry definition");
309
310 if (!ginput) {
311 egsWarning("EB_Application::createGeometry: no geometry specification in this input\n");
312 egsFatal("Simulation stopped\n");
313 }
314
315 int err = ginfo.initializeFromInput(ginput);
316 if (err) {
317 egsWarning("EB_Application::createGeometry: incomplete or incorrect geometry specification\n");
318 delete ginput;
319 egsFatal("Simulation stopped\n");
320 }
321
322 err = EGS_AdvancedApplication::initGeometry();
323
324 if (err) {
325 egsWarning("Failed to create a geometry from input file\n");
326 delete ginput;
327 egsFatal("Simulation stopped\n");
328 }
329
330 geometry->ref();
331
332 ginfo.setGeometryIndexes(geometry);
333
334
335
336 if (run_mode == RM_SUPERPOSITION) {
337 if (ginfo.source_envelope_name == "") {
338 egsFatal("Missing 'source envelope geometry' input required for superposition mode\n");
339 }
340
341 source_envelope_geom = EGS_BaseGeometry::getGeometry(ginfo.source_envelope_name);
342 if (!source_envelope_geom) {
343 egsFatal("source envelope geometry '%s' can not be found\n", ginfo.source_envelope_name.c_str());
344 }
345
346 if (source_envelope_geom->getType() != "EGS_ASwitchedEnvelope") {
347 egsFatal("You must use an %s source envelope geometry type for superposition mode\n", "EGS_ASwitchedEnvelope");
348 }
349 superpos_geom = static_cast<EGS_ASwitchedEnvelope *>(source_envelope_geom);
350 }else{
351 for (int gg=0; gg < ginfo.ngeom; gg++) {
352 GeomRegionInfo gr = ginfo.ordered_geom_data[gg];
353 if (gr.type == "EGS_ASwitchedEnvelope"){
354 egsFatal("EGS_ASwitchedEnvelope should not be used when not using 'run mode = superposition\n");
355 }
356 }
357 }
358
359
360 err = initSourceTransforms();
361 if (err) {
362 egsFatal("Failed to initiate source locations\n");
363 }
364
365 delete ginput;
366
367 err = createPhantoms();
368 if (err) {
369 egsWarning("Failed to create phantom objects\n");
370 egsFatal("Simulation stopped\n");
371 }
372
373 err = correctVolumes();
374
375 timing_blocks.stopTimer();
376 return 0;
377

443int EB_Application::checkSourceOverlaps(EGS_Input *inp) {
444
445 timing_blocks.addTimer("egs_brachy::checkSourceOverlaps");
446
447 vector<string> yn_choices;
448 yn_choices.push_back("no");
449 yn_choices.push_back("yes");
450 bool check_source_overlap = (bool)inp->getInput("check source overlaps", yn_choices, 0);
451
452 if (!check_source_overlap || source_transforms.size() <= 1){
453 /* not requested or only 1 source so no need to check for overlap */
454 timing_blocks.stopTimer();
455 return 0;
456 }
457
458 vector<string> mode_choices;
459 mode_choices.push_back("warning");
460 mode_choices.push_back("fatal");
461
462 bool fatal = (bool)inp->getInput("warning mode", mode_choices, 1);
463
464 EGS_Input *shape_inp = inp->takeInputItem("shape");
465
466 EGS_BaseShape *bounds = EGS_BaseShape::createShape(shape_inp);
467
468 if (!shape_inp) {
469 egsWarning("egs_brachy::checkSourceOverlaps - no `shape` input found.\n");
470 return 1;
471 }
472
473 EGS_Float bounds_volume = ebvolcor::getShapeVolume(shape_inp);
474 if (bounds_volume < 0){
475 egsWarning("egs_brachy::checkSourceOverlaps - Unable to get shape volume.");
476 return 1;
477 }
478
479 vector<string> excluded;
480 inp->getInput("excluded geometries", excluded);
481
482 EGS_Float density;
483 int err = inp->getInput("density of random points (cm^-3)", density);
484 if (err) {
485 egsWarning("egs_brachy::checkSourceOverlaps - The volume correction 'density of random points (cm^-3)' input was not found. Using 1E6/cm^3\n");
486 density = 1E6;
487 }
488 EGS_I64 npoints = (EGS_I64)floor(max(1., density*bounds_volume));
489
490
491 EGS_RandomGenerator *rng = EGS_RandomGenerator::defaultRNG();
492 EGS_Vector point;
493
494 /* find first source geometry defined */
495 EGS_BaseGeometry *base_source = 0;
496 string base_source_name;
497 for (int gg=0; gg < ginfo.ngeom; gg++) {
498 GeomRegionInfo gr = ginfo.ordered_geom_data[gg];
499 if (find(ginfo.source_names.begin(), ginfo.source_names.end(), gr.name) != ginfo.source_names.end()){
500 base_source = EGS_BaseGeometry::getGeometry(gr.name);
501 base_source_name = gr.name;
502 break;
503 }
504 }
505 if (!base_source){
506 egsFatal("egs_brachy::checkSourceOverlaps - did not find base source geometry");
507 }
508
509 /* now lets see if our base source is in an autoenvelope */
510 EGS_AffineTransform base_transform;
511 EGS_AffineTransform inv_base_transform;
512 for (int gg=0; gg < ginfo.ngeom; gg++) {
513 GeomRegionInfo gr = ginfo.ordered_geom_data[gg];
514 if (find(gr.children.begin(), gr.children.end(), base_source_name) != gr.children.end()){
515 /* base source is child of this geometry */
516 if (gr.type != "EGS_AEnvelope" && gr.type != "EGS_ASwitchedEnvelope"){
517 base_transform = *source_transforms[0];
518 inv_base_transform = base_transform.inverse();
519 }
520 break;
521 }
522 }
523
524 vector<int> overlaps;
525
526 for (EGS_I64 i=0; i < npoints; i++) {
527
528 /* Generate a point and check if its contained within base source. If
529 * it's not actually in a source, go back and generate a new point. */
530 point = bounds->getRandomPoint(rng);
531 base_transform.transform(point);
532 if (base_source->isWhere(point) < 0) {
533 continue;
534 }
535 inv_base_transform.transform(point);
536
537 for (size_t sa_idx = 0; sa_idx < source_transforms.size(); sa_idx++){
538
539 overlaps.clear();
540
541 EGS_Vector transformed(point);
542
543 // transform from point relative to origin to point
544 // relative to source A we are checking against other sources (B)
545 source_transforms[sa_idx]->transform(transformed);
546
547 for (size_t sb_idx = sa_idx + 1; sb_idx < source_transforms.size(); sb_idx++){
548
549 EGS_Vector inner_transformed(transformed);
550
551 // use current source we are checking transform to back relative to origin
552 source_transforms[sb_idx]->inverse().transform(inner_transformed);
553 base_transform.transform(inner_transformed);
554 if (base_source->isWhere(inner_transformed) >= 0) {
555 /* point falls within Source A & B so they must be overlapping */
556 overlaps.push_back(sa_idx);
557 overlaps.push_back(sb_idx);
558 goto overlap_found;
559 }
560 }
561 }
562 }
563
564overlap_found:
565
566 if (rng){
567 delete rng;
568 }
569 if (shape_inp){
570 delete shape_inp;
571 }
572 timing_blocks.stopTimer();
573
574 if (overlaps.size() > 1){
575 string msg = "Possible overlap of sources: ";
576 for (int i=0; i < overlaps.size(); i++){
577 msg += to_string(overlaps[i]);
578 if (i != overlaps.size() -1 ){
579 msg += ", ";
580 }
581 }
582 egsInformation((msg+"\n").c_str());
583 }
584
585 if (overlaps.size() > 1){
586 return fatal ? 1 : 0;
587 }
588
589 return 0;
590}

592int EB_Application::correctVolumes() {
593
594
595 timing_blocks.addTimer("egs_brachy::correctVolumes");
596 EGS_Input *vol_cor_inp = input->takeInputItem("volume correction");
597 if (!vol_cor_inp) {
598 egsWarning("Input item `volume correction` was not found\n");
599 return 1;
600 }
601
602 if (run_mode == RM_SUPERPOSITION) {
603 for (int i=0; i < nsources; i++) {
604 superpos_geom->activateByIndex(i);
605 }
606 }
607 ebvolcor::VolumeCorrector vc(vol_cor_inp, phantom_geoms, geometry, &ginfo, source_transforms);
608
609 source_vc_results = vc.runSourceCorrection(timing_blocks);
610
611 gen_vc_results = vc.runGeneralCorrection(timing_blocks);
612
613 file_vc_results = vc.runFileCorrection(timing_blocks);
614
615
616 // automatic volumes now done, now get any manually specified volumes
617 EGS_Input *ij;
618 vector<string> user_vols;
619
620 while ((ij = vol_cor_inp->takeInputItem("phantom region volumes")) != 0) {
621
622 string phant_name;
623 ij->getInput("phantom name", phant_name);
624 EB_Phantom *phant = getPhantomByName(phant_name);
625 if (!phant) {
626 egsFatal("`phanton region volume` specified for phantom `%s` which does not exist.", phant_name.c_str());
627 }else{
628 user_vols.push_back(phant_name);
629 }
630
631 vector<int> phantom_regs;
632 ij->getInput("region numbers", phantom_regs);
633
634 vector<EGS_Float> phantom_vols;
635 ij->getInput("region volumes", phantom_vols);
636
637 if (phantom_regs.size() != phantom_vols.size()){
638 egsFatal(
639 "Mismatched number of inputs for `region numbers` and `region volumes` for phantom `%s`",
640 phant_name.c_str()
641 );
642 }else if (phantom_regs.size() == 0){

1629int EB_Application::ausgab(int iarg) {
1630
1631 bool is_photon = top_p.q == 0;
1632 bool is_before_transport = iarg == BeforeTransport;
1633 bool is_after_transport = iarg == AfterTransport;
1634
1635 if (is_after_transport && isStuck()) {
1636 the_stack->latch[the_stack->np - 1] = top_p.latch;
1637 return 1;
1638 }
1639
1640 int global_ir = top_p.ir;
1641 int irnew = the_epcont->irnew-2;
1642 int irold = the_epcont->irold-2;
1643
1644 bool in_phantom = ginfo.isPhantom(global_ir);
1645 bool in_source = ginfo.isSource(global_ir);
1646
1647 bool is_extra_scoring_reg = false;
1648 string local_geom_name = "";
1649 int local_geom_ir = -1;
1650 int extra_reg_dose_index = -1;
1651
1652 if (extra_scoring_reg.size() > 0 && global_ir >= 0){
1653
1654 // local geom object info
1655 GeomRegT local = ginfo.globalToLocal(global_ir);
1656 local_geom_ir = local.second;
1657 local_geom_name = local.first->getName();
1658
1659 // check if the current geometry has extra scoring regions
1660 map<string, vector<int> >::iterator i = extra_scoring_reg.find(local_geom_name);
1661 if (i != extra_scoring_reg.end()){
1662 vector<int>::iterator it = find(i->second.begin(), i->second.end(), local_geom_ir);
1663 if (it != i->second.end()){
1664 is_extra_scoring_reg = true;
1665 extra_reg_dose_index = distance(i->second.begin(), it);
1666 }
1667 }
1668 }
1669
1670 /* track number of steps */
1671 if (is_before_transport) {
1672 if (in_source) {
1673 steps_in_sources[top_p.q]++;
1674 } else if (in_phantom) {
1675 steps_in_phantoms[top_p.q]++;
1676 } else {
1677 steps_in_other[top_p.q]++;
1678 }
1679 }
1680
1681 /* Fluorescent photon cutoff */
1682 bool discard_fluorescent = iarg == FluorescentEvent && top_p.E <= flu_cutoff;
1683 if (discard_fluorescent) {
1684 discardTopParticle();
1685 the_stack->latch[the_stack->np - 1] = top_p.latch;
1686 return 0;
1687 }
1688
1689 /* track scattering with latch bits if required */
1690 if (score_scat && (iarg > AfterTransport) && !in_source) {
1691 latch_control.addScatter(&top_p);
1692 for (int ip=the_stack->npold-1; ip < the_stack->np; ip++) {
1693 latch_control.addScatter(the_stack->latch[ip]);
1694 }
1695 }
1696
1697 bool next_is_source = ginfo.isSource(irnew);
1698 bool last_was_source = ginfo.isSource(irold);
1699
1700 /* note a particle leaving source could be a particle that has re-entered a
1701 * source. escaping_source will only be true on first time leaving source*/
1702 bool leaving_source = is_before_transport && in_source && !next_is_source;
1703 bool escaping_source = leaving_source && !latch_control.hasEscaped(&top_p);
1704
1705 /* same as leaving/escaping but for after transport rather than before*/
1706 bool left_source = is_after_transport && !in_source && last_was_source;
1707 bool escaped_source = left_source && !latch_control.hasEscaped(&top_p);
1708
1709 bool escaping_geom = is_before_transport && irnew < 0;
1710 bool escaped_geom = global_ir < 0;
1711
1712
1713 /* these notifications may be used by subscribers like spectrum
1714 * scoring objects, latch control etc */
1715 EBSendMessage send_messages[] = {
1716 EBSendMessage(is_before_transport, PARTICLE_TAKING_STEP),
1717 EBSendMessage(is_after_transport, PARTICLE_TOOK_STEP),
1718 EBSendMessage(escaping_source, PARTICLE_ESCAPING_SOURCE),
1719 EBSendMessage(escaped_source, PARTICLE_ESCAPED_SOURCE),
1720 EBSendMessage(escaping_geom, PARTICLE_ESCAPING_GEOM),
1721 EBSendMessage(escaped_geom, PARTICLE_ESCAPED_GEOM)
1722 };
1723 int nmesg = sizeof(send_messages)/sizeof(send_messages[0]);
1724 for (int m=0; m < nmesg; m++) {
1725 if (send_messages[m].first) {
1726 pevent_pub.notify(send_messages[m].second, &top_p);
1727 }
1728 }
1729
1730 /* copy the top_p.latch bit to the stack before returning from ausgab.
1731 * Since top_p.latch gets reset every step */
1732 the_stack->latch[the_stack->np - 1] = top_p.latch;
1733
1734 if (escaped_geom) {
1735 return 0;
1736 }
1737
1738 /* move particles to new locations if single generator or recycling */
1739 if (escaped_source && is_after_transport) {
1740
1741 EGS_Float wt = top_p.wt;
1742
1743 if (run_mode == RM_SUPERPOSITION) {
1744 superpos_geom->setActiveByIndex(active_source);
1745 }
1746
1747 if (recycle_opts) {
1748 addRecycledParticlesToStack(&top_p);
1749 } else if (single_generator && active_source > 0) {
1750 copyParticleToSourceLoc(&top_p, active_source, true, false, wt);
1751 }
1752
1753 return 0;
1754
1755 }
1756
1757 /* handle bcse and splitting */
1758 doPhotonSplitting(iarg);
1759
1760
1761 /* And finally below here we handle dose scoring */
1762 bool in_vaccuum = the_useful->medium <= 0;
1763 bool score_tracklength = score_tlen && is_photon && is_before_transport;
1764 bool score_interaction = score_edep && iarg <= ExtraEnergy;
1765 bool dose_scoring_not_needed = !is_extra_scoring_reg && (!in_phantom || in_vaccuum || !(score_tracklength || score_interaction));
1766
1767 if (dose_scoring_not_needed) {
1768 return 0;
1769 }
1770
1771
1772 // which phantom/region are we in
1773 int phant_idx = ginfo.phantomFromRegion(global_ir);
1774 int phant_ir = ginfo.globalToLocalReg(global_ir);
1775
1776 EB_Phantom *phant = phantom_geoms[phant_idx];
1777
1778 bool needs_vol_cor = run_mode == RM_SUPERPOSITION && superpos_geom->hasInactiveGeom(global_ir);
1779
1780
1781 if (score_tracklength) {
1782
1783 EGS_Interpolator *interp = media_muen[the_useful->medium-1];
1784 EGS_Float muen_val = interp ? interp->interpolateFast(the_epcont->gle) : 0;
1785 EGS_Float tracklength_edep = the_epcont->tvstep*top_p.E*muen_val*top_p.wt;
1786
1787 EGS_Float vol;
1788
1789 if (in_phantom){
1790 if (needs_vol_cor) {
1791 vol = phant->getUncorrectedVolume(phant_ir);
1792 } else {
1793 vol = phant->getCorrectedVolume(phant_ir);
1794 }
1795
1796 if (vol > 0) {
1797 phant->scoreTlen(phant_ir, tracklength_edep / vol, &top_p);
1798 }
1799 }
1800
1801 if (is_extra_scoring_reg){
1802 vol = extra_scoring_vols[local_geom_name][extra_reg_dose_index];
1803 extra_scoring_doses[local_geom_name]->score(extra_reg_dose_index, tracklength_edep / vol);
1804 }
1805 }
1806
1807 if (score_interaction) {
1808
1809 EGS_Float mass;
1810
1811 EGS_Float edep = getEdep()*top_p.wt;
1812
1813 if (in_phantom){
1814
1815 if(needs_vol_cor) {
1816 mass = phant->getUncorrectedMass(phant_ir);
1817 } else {
1818 mass = phant->getRealMass(phant_ir);
1819 }
1820
1821 if (mass > 0) {
1822 phant->scoreEdep(phant_ir, edep / mass);
1823 }
1824 }
1825
1826 if (is_extra_scoring_reg){
1827 mass = extra_scoring_mass[local_geom_name][extra_reg_dose_index];
1828 extra_scoring_doses[local_geom_name]->score(extra_reg_dose_index, edep / mass);
1829 }
1830 }
1831
1832 return 0;
1833}

1835void EB_Application::doPhotonSplitting(int iarg) {
1836
1837 bool is_phat = fabs(top_p.wt - 1.) < EB_EPSILON;
1838
1839 if (do_bcse) {
1840
1841 bool in_bcse_med = the_useful->medium - 1 == bcse_med_num;
1842
1843 if (iarg == BeforeBrems) {
1844 the_egsvr->nbr_split = do_brem_split ? nbr_split : 1;
1845 if (!in_bcse_med) {
1846 /* standard UBS for brem photons OUTSIDE MED_BCSE */
1847 the_egsvr->nbr_split *= (int)bcse_factor;
1848 }
1849 } else if (iarg == AfterBrems) {
1850 if (in_bcse_med) {
1851 /* Just came back from a brem event in MED_BCSE, reduce photon weight
1852 and play Russian Roulette to whether or not reduce electron energy */
1853 for (int ip=the_stack->npold; ip < the_stack->np; ip++) {
1854 the_stack->wt[ip] /= bcse_factor;
1855 }
1856 EGS_Float rnd = rndm->getUniform();
1857 if (rnd > 1./bcse_factor) {
1858 the_stack->E[the_stack->npold-1] += top_p.E;
1859 }
1860 } else {
1861 /* Turn off UBS after splitting that one fat brem photons outside MED_BCSE */
1862 the_egsvr->nbr_split = 1;
1863 }
1864
1865 } else if (iarg == FluorescentEvent && is_phat) {
1866
1867 /* Uniformly splits fat relaxation photons that are descendents of either:
1868 1. eii in MED_BCSE (usually MED_BCSE is the target) or, 2.
1869 back-scattered fat electrons that escaped target and interacted
1870 elsewhere (eii and photo) This is done using UNIFORM_PHOTONS */
1871
1872 int nsplit = do_brem_split ? (int)(bcse_factor*nbr_split) : (int)(bcse_factor);
1873 the_stack->npold = the_stack->np;
1874 F77_OBJ_(egs_uniform_photons, EGS_UNIFORM_PHOTONS)(&nsplit, &top_p.E);
1875
1876 }
1877
1878 } else if (do_brem_split && iarg == FluorescentEvent && is_phat) {
1879
1880 /* a fluorescent photon has just been put on the stack - split it */
1881 the_stack->npold = the_stack->np;
1882 F77_OBJ_(egs_uniform_photons, EGS_UNIFORM_PHOTONS)(&nbr_split, &top_p.E);
1883 }
1884
1885}
